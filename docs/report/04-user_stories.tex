\FloatBarrier\section{User Stories}\label{sec:user-stories}
The numbering below corresponds to the indexes of the User Stories defined in the project documentation.

\vskip 0.5cm

\paragraph{Account}
\begin{enumerate}[start=2]
    \item \api{POST /register} \\
        Gateway $\longrightarrow$ Authentication $\longrightarrow$ \textit{Users} $\longrightarrow$ Authentication $\longrightarrow$ Gateway

    \item \api{POST /login} \\
        Gateway $\longrightarrow$ Authentication $\longrightarrow$ \textit{Users} $\longrightarrow$ Authentication $\longrightarrow$ Gateway

    \item
    \begin{enumerate}
        \item \api{GET /players/me} \\
            Gateway $\longrightarrow$ Players $\longrightarrow$ \textit{Players} $\longrightarrow$ Players $\longrightarrow$ Gateway
        \item \api{PATCH /players/me} \\
            Gateway $\longrightarrow$ Players $\longrightarrow$ \textit{Players} $\longrightarrow$ Players $\longrightarrow$ Gateway
    \end{enumerate}
\end{enumerate}

\paragraph{Cards}
\begin{enumerate}[start=6]
    \item \api{GET /cards} \\
        Gateway $\longrightarrow$ Cards $\longrightarrow$ \textit{Cards} $\longrightarrow$ Cards $\longrightarrow$ Gateway
    \item \api{GET /cards/<card\_id>} \\
        Gateway $\longrightarrow$ Cards $\longrightarrow$ \textit{Cards} $\longrightarrow$ Cards $\longrightarrow$ Gateway
\end{enumerate}

\paragraph{Game}
\begin{enumerate}[start=8]
    \item
    \begin{enumerate}
        \item \api{POST /enqueue} \\
            Gateway $\longrightarrow$ Matchmaking $\longrightarrow$ Players $\longrightarrow$ \textit{Players} $\longrightarrow$ Players $\longrightarrow$ Matchmaking $\longrightarrow$ \textit{Queue} $\longrightarrow$ Matchmaking $\dashrightarrow$ Game Engine $\dashrightarrow$ \textit{Matches} $\dashrightarrow$ Game Engine $\dashrightarrow$ Matchmaking $\longrightarrow$ Gateway
        \item \api{POST /internal/matches/create} \\
            Gateway $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Matches} $\longrightarrow$ Game Engine $\longrightarrow$ Gateway
    \end{enumerate}

    \item \api{POST /matches/<match\_id>/deck} \\
        Gateway $\longrightarrow$ Game Engine $\longrightarrow$ Card Catalogue $\longrightarrow$ \textit{Cards} $\longrightarrow$ Card Catalogue $\longrightarrow$ Game Engine $\longrightarrow$ Gateway
        
    \item \api{POST /matches/<match\_id>/moves/<round\_id>} \\
        Gateway $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Matches} $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Rounds} $\longrightarrow$ Game Engine $\longrightarrow$ Gateway

    \item \api{GET /matches/<match\_id>} \\
        Gateway $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Matches} $\longrightarrow$ Game Engine $\longrightarrow$ Gateway

    \item Know the turns.

    \item \api{GET /matches/<match\_id>} \\
        Gateway $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Matches} $\longrightarrow$ Game Engine $\longrightarrow$ Gateway

    \item \api{GET /matches/<match\_id>/round} \\
        Gateway $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Matches} $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Rounds} $\longrightarrow$ Game Engine $\longrightarrow$ Gateway
    
    \item 
    \begin{enumerate}
        \item \api{GET /matches/<match\_id>} \\
            Gateway $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Matches} $\longrightarrow$ Game Engine $\longrightarrow$ Gateway
        \item \api{GET /matches/<match\_id>/history} \\
            Gateway $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Matches} $\longrightarrow$ Game Engine $\longrightarrow$ \textit{Rounds} $\longrightarrow$ Game Engine $\longrightarrow$ Gateway
    \end{enumerate}
\end{enumerate}
